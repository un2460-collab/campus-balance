  <script>
    const STORAGE_KEY = "campusBalanceEvents";

    const categoryConfig = {
      study: {
        label: "학업",
        color: "#3b82f6",
        fatigue: 1
      },
      exam: {
        label: "과제·시험",
        color: "#ef4444",
        fatigue: 3
      },
      work: {
        label: "알바",
        color: "#facc15",
        textColor: "#1f2937",
        fatigue: 2
      },
      private: {
        label: "사적 약속",
        color: "#22c55e",
        fatigue: -1
      },
      etc: {
        label: "기타",
        color: "#9ca3af",
        fatigue: 0
      }
    };

    let calendar;
    let selectedDate = getTodayString();

    function getTodayString() {
      const today = new Date();
      return formatDate(today);
    }

    function formatDate(date) {
      const year = date.getFullYear();
      const month = String(date.getMonth() + 1).padStart(2, "0");
      const day = String(date.getDate()).padStart(2, "0");
      return `${year}-${month}-${day}`;
    }

    function loadEvents() {
      const saved = localStorage.getItem(STORAGE_KEY);
      return saved ? JSON.parse(saved) : [];
    }

    function saveEvents(events) {
      localStorage.setItem(STORAGE_KEY, JSON.stringify(events));
    }

    function getCalendarEvents() {
      return loadEvents().map(event => {
        const config = categoryConfig[event.category];

        return {
          id: event.id,
          title: `[${config.label}] ${event.title}`,
          start: event.start,
          end: event.end,
          backgroundColor: config.color,
          borderColor: config.color,
          textColor: config.textColor || "#ffffff",
          extendedProps: {
            category: event.category,
            originalTitle: event.title
          }
        };
      });
    }

    function calculateFatigue(dateString) {
      const events = loadEvents();
      let score = 0;

      events.forEach(event => {
        const eventDate = event.start.slice(0, 10);

        if (eventDate === dateString) {
          score += categoryConfig[event.category].fatigue;
        }
      });

      return score;
    }

    function updateDashboard(dateString) {
      selectedDate = dateString;

      const score = calculateFatigue(dateString);
      const fatigueScore = document.getElementById("fatigueScore");
      const selectedDateLabel = document.getElementById("selectedDateLabel");
      const alertBanner = document.getElementById("alertBanner");
      const aiMessage = document.getElementById("aiMessage");

      fatigueScore.textContent = score;
      selectedDateLabel.textContent = `${dateString} 기준`;

      if (score > 7) {
        alertBanner.classList.remove("hidden");
        fatigueScore.className = "text-4xl font-black text-red-600";
        aiMessage.textContent =
          "오늘 학업과 알바로 일과 로드가 높습니다. 빈 시간에는 추가 일정을 잡지 말고 휴식을 취하세요!";
      } else if (score >= 4) {
        alertBanner.classList.add("hidden");
        fatigueScore.className = "text-4xl font-black text-orange-500";
        aiMessage.textContent =
          "조금 바쁜 하루예요. 중요한 일정 사이에 짧은 휴식 시간을 확보해 보세요.";
      } else if (score >= 1) {
        alertBanner.classList.add("hidden");
        fatigueScore.className = "text-4xl font-black text-blue-600";
        aiMessage.textContent =
          "무난한 일정이에요. 집중 시간과 휴식 시간을 균형 있게 유지해 보세요.";
      } else {
        alertBanner.classList.add("hidden");
        fatigueScore.className = "text-4xl font-black text-green-600";
        aiMessage.textContent =
          "여유가 있는 하루예요. 재충전하거나 개인 시간을 보내기 좋아요.";
      }
    }

    function refreshCalendar() {
      calendar.removeAllEvents();
      getCalendarEvents().forEach(event => calendar.addEvent(event));
      updateDashboard(selectedDate);
    }

    function addEvent(eventData) {
      const events = loadEvents();
      events.push(eventData);
      saveEvents(events);
      refreshCalendar();
    }

    function deleteEvent(eventId) {
      const events = loadEvents().filter(event => event.id !== eventId);
      saveEvents(events);
      refreshCalendar();
    }

    function setDefaultDateTimeInputs(dateString) {
      const startInput = document.getElementById("start");
      const endInput = document.getElementById("end");

      startInput.value = `${dateString}T09:00`;
      endInput.value = `${dateString}T10:00`;
    }

    document.addEventListener("DOMContentLoaded", function () {
      const calendarEl = document.getElementById("calendar");

      calendar = new FullCalendar.Calendar(calendarEl, {
        initialView: "dayGridMonth",
        locale: "ko",
        height: "auto",
        selectable: true,
        nowIndicator: true,
        headerToolbar: {
          left: "prev,next today",
          center: "title",
          right: "dayGridMonth,timeGridWeek,timeGridDay,listWeek"
        },
        buttonText: {
          today: "오늘",
          month: "월",
          week: "주",
          day: "일",
          list: "목록"
        },
        events: getCalendarEvents(),

        dateClick: function (info) {
          updateDashboard(info.dateStr);
          setDefaultDateTimeInputs(info.dateStr);
        },

        eventClick: function (info) {
          const confirmDelete = confirm(
            `"${info.event.title}" 일정을 삭제할까요?`
          );

          if (confirmDelete) {
            deleteEvent(info.event.id);
          }
        },

        eventDidMount: function (info) {
          info.el.title = "클릭하면 삭제할 수 있어요.";
        }
      });

      calendar.render();

      setDefaultDateTimeInputs(selectedDate);
      updateDashboard(selectedDate);

      document.getElementById("eventForm").addEventListener("submit", function (e) {
        e.preventDefault();

        const title = document.getElementById("title").value.trim();
        const category = document.getElementById("category").value;
        const start = document.getElementById("start").value;
        const end = document.getElementById("end").value;

        if (!title || !category || !start || !end) {
          alert("모든 항목을 입력해 주세요.");
          return;
        }

        if (new Date(start) >= new Date(end)) {
          alert("종료 시간은 시작 시간보다 늦어야 합니다.");
          return;
        }

        const newEvent = {
          id: crypto.randomUUID(),
          title,
          category,
          start,
          end
        };

        addEvent(newEvent);
        updateDashboard(start.slice(0, 10));

        e.target.reset();
        setDefaultDateTimeInputs(start.slice(0, 10));
      });
    });
  </script>
